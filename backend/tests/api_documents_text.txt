// 1. جلب كل الملفات مع الفلاتر الديناميكية (Index)
// نوع الطلب: GET
// الرابط مع الفلاتر:
// http://127.0.0.1:8000/api/library/documents?search=report&sort_by=title&show=expired_on&expired_on_date=2024-12-31&show_at_least=25


// 2. تغيير حالة ملفات (Bulk Action: deleted, suspended, active)
// نوع الطلب: POST
// الرابط: http://127.0.0.1:8000/api/library/documents/action
// الـ Body (JSON):
{
  "action": "suspended",
  "document_ids": [1, 2, 5]
}


// 3. جلب تفاصيل ملف واحد (Show)
// نوع الطلب: GET
// الرابط: http://127.0.0.1:8000/api/library/documents/1
// (الرقم 1 هو ID الملف)


// 4. تعديل بيانات ملف (Update Note & Expiry Date)
// نوع الطلب: PUT
// الرابط: http://127.0.0.1:8000/api/library/documents/1
// الـ Body (JSON):
{
  "note": "تم تحديث الوصف لهذا الملف الهام",
  "expiry_date": "2025-01-01"
}


// 5. جلب الملفات التابعة لمنشور محدد (Publication Documents)
// نوع الطلب: GET
// الرابط: http://127.0.0.1:8000/api/library/publications/10/documents
// (الرقم 10 هو ID المنشور)
